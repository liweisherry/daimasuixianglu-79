```
class Solution(object):
    op_map = {
        '+': add,
        '-': sub,
        '*': mul,
        '/': lambda a, b: int(a / b)  # truncate toward zero
    }

    def evalRPN(self, tokens: List[str]) -> int:
        stack = []
        for token in tokens:
            if token not in self.op_map:
                stack.append(int(token))
            else:
                op2 = stack.pop()
                op1 = stack.pop()
                stack.append(self.op_map[token](op1, op2))
        return stack.pop()
```
