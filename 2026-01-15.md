```
    def minSubArrayLen(self, target: int, nums: List[int]) -> int:
        p1, p2 = 0, 0
        res = float('inf')
        sums = 0
        length = 0
        while p2 < len(nums):
            sums += nums[p2]
            while sums >= target:
                length = p2 - p1 + 1
                res = length if length < res else res 
                sums -= nums[p1]
                p1 += 1
            p2 += 1
        return res if res != float('inf')  else 0
```
