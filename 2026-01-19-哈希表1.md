    def isAnagram(self, s: str, t: str) -> bool:

            return False
        check = {}
        check_2 = {}
        for w in s:
            if w in check:
                check[w] +=1
            else:
                check[w] = 1
        for w in t:
            if w in check_2:
                check_2[w] +=1
            else:
                check_2[w] = 1
        return check == check_2
