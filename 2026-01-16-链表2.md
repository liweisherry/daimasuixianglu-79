```
      def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        
        dummy = ListNode(next=head)
        cur = dummy
        while cur.next and cur.next.next:
            p1 = cur.next
            p3 = cur.next.next.next
            cur.next = cur.next.next
            cur.next.next = p1
            cur.next.next.next = p3
            cur = cur.next.next
        return dummy.next
```
